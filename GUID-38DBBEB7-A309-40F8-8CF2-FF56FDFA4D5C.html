Example: Application Processing for Revolving Category

Use a base application and test scenarios to evaluate policies designed for processing applications for a product category.
The following base application tests different components and strategies designed for a product category called Revolving: 
<Application DeliveryOptionCode=""build251"" ProcessingRequestType=""DM"">"						
	"<CreditRequest ProductCategory=""Revolving""/>"					
	<productCategory>					
		<coAppEvaluationPolicy>EVALUATION_POLICY_TYPE_BOTH_APPLICANTS_TOGETHER</coAppEvaluationPolicy>				
	</productCategory>					
	<creditApplication>					
		<decisionParty>DECISION_PARTY_TYPE_PRIMARY</decisionParty>				
		<type>OM_CREDIT_APPLICATION_TYPE_INDIVIDUAL</type>				
		<enteredTimestamp>2014-09-18T19:48:50</enteredTimestamp>				
		<appNumber>0</appNumber> 				
		<booking>				
			<creditDetails>			
				<creditApplicationProduct>		
					<creditApplicationProductFee>	
						<feeAmount>0</feeAmount>
					</creditApplicationProductFee>	
				</creditApplicationProduct>		
			</creditDetails>			
		</booking>				
		<personalApplicant>				
			<age/>			
			<monthsWithPresentEmployer/>			
			<person>			
				<birthDate>1980-09-18</birthDate>		
			</person>			
			<personalCraReportRequest>			
				<personalCraReportResponse>		
					<craScoring>	
						<score>450</score>
					</craScoring>	
					<personalCraCharacteristic>	
						<securityReportCode>00</securityReportCode>
						<addressMismatchFlag>false</addressMismatchFlag>
						<aliasFlag>false</aliasFlag>
						<deceasedFlag>false</deceasedFlag>
						<nameMismatchFlag>false</nameMismatchFlag>
						<ssnVariationFlag>false</ssnVariationFlag>
						<noRecordAtCraFlag>false</noRecordAtCraFlag>
						<worstRatingCode>E9</worstRatingCode>
						<bankruptcyOnFileFlag>false</bankruptcyOnFileFlag>
						
					</personalCraCharacteristic>	
				</personalCraReportResponse>		
			</personalCraReportRequest>			
			<currentEmployment>			
				<employment>		
				<employmentType>OM_EMPLOYMENT_TYPE_1</employmentType>		
					<isSelfEmployed>yes</isSelfEmployed>	
					<startDate>2010-02-01</startDate>	
					<monthsWithEmployer>56</monthsWithEmployer>	
				</employment>		
			</currentEmployment>			
			<applicantFinancial>			
				<totalGrossMonthlyIncome>6000</totalGrossMonthlyIncome>		
				<income>		
					<grossMonthlyIncome>1000</grossMonthlyIncome>	
				</income>		
			</applicantFinancial>			
		</personalApplicant>				
	</creditApplication>					
</Application>
